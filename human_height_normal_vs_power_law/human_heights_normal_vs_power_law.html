def normfun(x,mu,sigma):
    pdf = np.exp(-((x - mu)**2)/(2*sigma**2)) / (sigma * np.sqrt(2*np.pi))
    return pdf

height_pdf = functools.partial(normfun, mu=data.mean(),sigma=data.std())
